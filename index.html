Question1. Find the Product.

Answer

    const Find_Prod = (array, N) => {
  let product = 1;
  for(let i = 0; i < N; i++){
    product *= array[i];
  }
  return product;
};

Question4. Even Odd

Answer

    const findEvenOdd = (N, Arr) => 
{
    let even = 0, odd = 0;
    for(let i = 0; i < N; i++){
      if(Arr[i]%2) odd+= Arr[i];
      else even += Arr[i];
    }
    return [even, odd];
};

Question11. Birthday Game

Answer

    function Birthday_Game(arr,D ,M) {
  let count = 0;
  for(let i = 0; i <= arr.length - M; i++){
    let sum = 0;
    for(let j = i; j < i + M; j++){
      sum += arr[j];
    }
    if(sum === D) count++;
  }
return count;
}
